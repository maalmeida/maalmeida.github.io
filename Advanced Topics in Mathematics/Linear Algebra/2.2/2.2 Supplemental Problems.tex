    \textbf{\S 2.2 The Inverse of a Matrix - Supplemental Problems}
\end{center}

\uplevel{Use the algorithm discussed in class to find the inverses of the following matrices, if they exist}
\begin{questions}
    \question $A = \mat{2 & 0 & 1 \\ -3 & -1 & 0 \\ -2 & 0 & 1}$
    \begin{solution}
        $A^{-1} = \mat{ 1/4 & 0 & -1/4 \\ -3/4 & -1 & 3/4 \\ 1/2 & 0 & 1/2}$
    \end{solution}

    \question $B = \mat{1 & 0 & -2 \\ -3 & 1 & 4 \\ 2 & -3 & 4}$
    \begin{solution}
        $B^{-1} = \mat{8 & 3 & 1 \\ 10 & 4 & 1 \\ 7/2 & 3/4 & 1/2}$
    \end{solution}

    \question $C = \mat{1 & 2 & 0 \\ 3 & -1 & 2 \\ -2 & 3 & -2}$
    \begin{solution}
        $C$ is not invertible as $\mat{A & I} \sim \mat{1 & 2 & 0 & 1 & 0 & 0 \\ 0 & -7 & 2 & -3 & 1 & 0 \\ 0 & 0 & 0 & -1 & 1 & 1}$
    \end{solution}

    \question Let $D = \mat{-1 & 2 & 0 \\ 0 & -2 & 4 \\ 1 & 2 & k}$. Find $k$ such that $D$ is not invertible.
    \begin{solution}
        $\mat{D & I} \sim \mat{1 & 0 & -4 & -1 & -1 & 0 \\ 0 & 1 & -2 & 0 & -1/2 & 0 \\ 0 & 0 & k-8 & 1 & 2 & 1}$ \newline 
        We cannot row reduce to the first three columns to $I_3$ if $k-8 = 0$ and so a value of $k=8$ maked $D$ non-invertible.
    \end{solution}
\end{questions}
