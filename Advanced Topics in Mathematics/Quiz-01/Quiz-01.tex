    \question Provide an $\epsilon-\delta$ proof of the following:
    \[ \lim_{x\rightarrow -2} 2x^2-5x-12 = 6 \]

    Let $\epsilon$ and suppose $0 < \left|x-\left(-2\right)\right| = |x+2| < \delta$ where $\delta = \textrm{min}\left(1,\frac{\epsilon}{15}\right)$. Then,
    \begin{align*}
        | f(x) - L | &= | (2x^2-5x-12) - 6 | \\
        &= |2x^2 - 5x-18| \\
        &= |2x^2 + 4x - 9x - 18| \\
        &= |2x(x+2) - 9(x+2) | \\
        &= |(2x-9)(x+2)| \\ 
        &= |x+2| \cdot |2x-9| \\
        &< \delta \cdot |2x-9|
    \end{align*}
    We attempt to place an upper bound on $|2x-9|$:

    \begin{center}
        \includegraphics{Problem 3.png}    
    \end{center}
    Assume $\delta \leq 1$. Then,
    
    \begin{align*}
        |x-(-2)| &\leq 1 \\
        |x+2| &\leq 1  \\
        -1 \leq x+2 &\leq 1 \\
        -3 \leq x &\leq -1 \\
        -6 \leq 2x &\leq -2 \\
        -15 \leq 2x-9 &\leq -11 \\
        |2x-9| &\leq 15
    \end{align*}

    Alternatively, again assuming $\delta \leq 1$
    \begin{align*}
        |2x-9| &= |2(x+2)-13| \\
        &\leq 2|x+2| + 13 \textrm{ (Triangle Inequality)} \\
        &\leq 2(1) + 13 \\
        &= 15
    \end{align*}
    
    Continuing from earlier,
    \begin{align*}
        \ldots &< \delta \cdot |2x-9| \\
        &\leq 15\delta \\
        &\leq 15 \left( \frac{\epsilon}{15}\right) \\
        &\leq \epsilon 
    \end{align*}
\end{questions}
